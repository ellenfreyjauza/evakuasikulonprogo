/* ========================================================= ADOPSI FITUR PROTOTYPE → KE SCRIPT GEOJSON ASLI ========================================================= TEMPATKAN SCRIPT INI DI BAWAH: - inisialisasi map - load polygon - load line geojson - load point geojson Script ini TIDAK pakai dummy line/polyline lagi. Semua route, filter, toggle, dan interaksi memakai layer GeoJSON asli yang sudah berhasil jalan. ========================================================= */ /* ========================================================= GLOBAL VARIABLE ========================================================= */ let userMarker = null; let customRouteLayer = null; let routeVisible = true; let shelterVisible = true; let riskVisible = true; let roadStatusMode = false; /* ========================================================= QUICK ACTION ========================================================= */ /* ===== LOKASI SAYA ===== */ document .getElementById("locateButton") .addEventListener("click", () => { if (!navigator.geolocation) { alert("Browser tidak mendukung GPS"); return; } navigator.geolocation.getCurrentPosition( position => { const lat = position.coords.latitude; const lng = position.coords.longitude; if (userMarker) { map.removeLayer(userMarker); } userMarker = L.circleMarker([lat, lng], { radius: 5, fillColor: "#2563eb", color: "#ffffff", weight: 1.5, fillOpacity: 1 }).addTo(map); userMarker .bindPopup("Lokasi Anda"); map.setView([lat, lng], 15); }, () => { alert("Lokasi gagal didapatkan"); } ); }); /* ===== RESET ===== */ document .getElementById("resetButton") .addEventListener("click", () => { map.setView( [-7.74, 110.18], 11 ); }); /* ========================================================= SEARCH ========================================================= */ document .getElementById("searchButton") .addEventListener("click", () => { const keyword = document .getElementById("searchInput") .value .toLowerCase(); if (keyword.includes("samigaluh")) { map.setView( [-7.67, 110.12], 14 ); } else if (keyword.includes("girimulyo")) { map.setView( [-7.80, 110.15], 14 ); } else if (keyword.includes("kalibawang")) { map.setView( [-7.71, 110.24], 14 ); } else { alert("Lokasi tidak ditemukan"); } }); /* ========================================================= COPY KOORDINAT ========================================================= */ document .getElementById("copyViewButton") .addEventListener("click", () => { const center = map.getCenter(); const text = ` Lat : ${center.lat.toFixed(6)} Lng : ${center.lng.toFixed(6)} `; navigator.clipboard.writeText(text); alert("Koordinat berhasil disalin"); }); /* ========================================================= FILTER LAYER ========================================================= */ /* ===== FILTER JALUR ===== */ document .getElementById("filterRouteButton") .addEventListener("click", () => { routeVisible = !routeVisible; if (routeVisible) { map.addLayer(lineLayer); } else { map.removeLayer(lineLayer); } }); /* ===== FILTER TITIK KUMPUL ===== */ document .getElementById("filterShelterButton") .addEventListener("click", () => { shelterVisible = !shelterVisible; if (shelterVisible) { map.addLayer(pointLayer); } else { map.removeLayer(pointLayer); } }); /* ===== FILTER RISIKO ===== */ document .getElementById("filterRiskButton") .addEventListener("click", () => { riskVisible = !riskVisible; if (riskVisible) { map.addLayer(polygonLayer); } else { map.removeLayer(polygonLayer); } }); /* ===== TAMPILKAN SEMUA ===== */ document .getElementById("filterAllButton") .addEventListener("click", () => { if (!map.hasLayer(lineLayer)) { map.addLayer(lineLayer); } if (!map.hasLayer(pointLayer)) { map.addLayer(pointLayer); } if (!map.hasLayer(polygonLayer)) { map.addLayer(polygonLayer); } }); /* ========================================================= MODE DARURAT ========================================================= */ document .getElementById("safeModeButton") .addEventListener("click", () => { polygonLayer.eachLayer(layer => { layer.setStyle({ fillOpacity: 0.55, opacity: 1 }); }); alert("Mode darurat aktif"); }); /* ========================================================= STATUS JALUR ========================================================= */ document .getElementById("toggleRoadStatusButton") .addEventListener("click", () => { roadStatusMode = !roadStatusMode; lineLayer.eachLayer(layer => { if (roadStatusMode) { layer.setStyle({ color: "#16a34a", dashArray: "8,6", weight: 3 }); } else { layer.setStyle({ color: "#2563eb", weight: 2.5, dashArray: null }); } }); }); /* ========================================================= RUTE EVAKUASI ========================================================= */ /* ===== BUAT RUTE ===== */ document .getElementById("buildRouteButton") .addEventListener("click", () => { let origin; const originMode = document .getElementById("routeOrigin") .value; if ( originMode === "user" && userMarker ) { origin = userMarker.getLatLng(); } else { origin = map.getCenter(); } /* ===== CARI TITIK TERDEKAT ===== */ let nearestPoint = null; let nearestDistance = Infinity; pointLayer.eachLayer(layer => { const target = layer.getLatLng(); const distance = map.distance(origin, target); if (distance < nearestDistance) { nearestDistance = distance; nearestPoint = target; } }); if (!nearestPoint) { alert("Titik kumpul tidak ditemukan"); return; } /* ===== HAPUS ROUTE LAMA ===== */ if (customRouteLayer) { map.removeLayer(customRouteLayer); } /* ===== ROUTE BARU ===== */ customRouteLayer = L.polyline( [ [origin.lat, origin.lng], [nearestPoint.lat, nearestPoint.lng] ], { color: "#16a34a", weight: 4, opacity: 0.95 } ) .addTo(map); map.fitBounds( customRouteLayer.getBounds(), { padding: [40, 40] } ); /* ===== RINGKASAN ===== */ const distanceKm = ( nearestDistance / 1000 ).toFixed(2); const duration = Math.ceil(distanceKm * 14); document .getElementById( "bestRouteSummary" ) .innerHTML = `

Rute berhasil dibuat

Jalur menuju titik kumpul terdekat berhasil dibuat.

${distanceKm} km Estimasi jarak
${duration} menit Estimasi waktu

Jalur aktif dan aman `; }); /* ========================================================= CETAK ========================================================= */ document .getElementById("printRouteButton") .addEventListener("click", () => { window.print(); }); /* ========================================================= NEAREST POINT ========================================================= */ document .getElementById("nearestButton") .addEventListener("click", () => { if (!userMarker) { alert("Aktifkan lokasi terlebih dahulu"); return; } let nearest = null; let nearestDistance = Infinity; pointLayer.eachLayer(layer => { const dist = map.distance( userMarker.getLatLng(), layer.getLatLng() ); if (dist < nearestDistance) { nearestDistance = dist; nearest = layer; } }); if (nearest) { map.setView( nearest.getLatLng(), 16 ); nearest.openPopup(); } });
